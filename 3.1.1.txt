void partyAgain(){
    repeat(10){
        turnLeft();
        pickBeeper();
        if(frontIsClear()){
            moveForward();
        }
        else if(!frontIsClear()){
            dropBeeper();
            turnAround();
        }
        moveParty();
        turnLeft();
        if(frontIsClear()){
            moveForward();
        }
    }
}

void moveParty(){
    if(frontIsClear()){
        moveForward();
        moveParty();
    }
    else if(!frontIsClear()){
        dropBeeper();
        turnAround();
    }
    moveForward();
}

void partyAgain () { //Hauptbefehl
    
    while (onBeeper()) { //Mit einem repeat-Befehl ist es nicht allgemein.
        turnLeft();
        pickBeeper();
        moveParty(); //Ein Beeper wird positioniert
        turnLeft();
        if (frontIsClear()) {
            moveForward();
        }
    }
}

void moveParty () {//FÃ¼r eine if-Schleife
    if (frontIsClear()){
        moveForward();
        moveParty(); //Speichert die gegangenen Schritte.
    }
    if (frontIsClear()) {
        moveForward();
    }
    else {
        //Hier wird der Beeper positioniert. Das kommt hier, damit er sich nicht mehr bewegt.
        turnAround();
        dropBeeper();
    }
}
